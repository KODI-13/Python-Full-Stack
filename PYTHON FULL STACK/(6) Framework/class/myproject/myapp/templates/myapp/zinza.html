Example 1 --> {%with age=2 %} {% if age > 18 %} {{age}}
{% else %} {{"Not Eligible"}}
{%endif%} {%endwith%} Example 2 --> {%with age=21 %} {% if age > 18 %} {{"Eligible"}}
